import { useData } from "nextra/data";

export const getStaticProps = async () => {
  const res = await fetch(
    "https://mainnet-beta.api.drift.trade/stats/perpMarketAccounts"
  );
  const data = await res.json();

  const perpMarginData = data.result
    .filter((market) => !("prediction" in market.contractType))
    .map((market, index) => {
      return {
        index,
        name: market.name
          .map((charCode) => String.fromCharCode(charCode))
          .join("")
          .trim(),
        marginRatioInitial: (market.marginRatioInitial / 1e2).toFixed(2) + "%",
        marginRatioMaintenance: (market.marginRatioMaintenance / 1e2).toFixed(2) + "%",
        imfFactor: (market.imfFactor / 1e6).toFixed(6),
      };
    });

  return {
    props: {
      ssg: {
        perpMarginData,
      },
    },
    revalidate: 3600,
  };
};


export const PerpMarginTable = () => {
  const { perpMarginData } = useData();

  const headings = [
    "Index",
    "Market",
    "Initial Margin Ratio",
    "Maintenance Margin Ratio",
    "IMF Factor",
  ];

  return (
    <table>
      <thead>
        <tr>
          {headings.map((heading) => (
            <th key={heading}>{heading}</th>
          ))}
        </tr>
      </thead>
      <tbody>
        {perpMarginData.map((row) => (
          <tr key={row.index}>
            <td>{row.index}</td>
            <td>{row.name}</td>
            <td>{row.marginRatioInitial}</td>
            <td>{row.marginRatioMaintenance}</td>
            <td>{row.imfFactor}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};
